CREATE TABLE `prueba1`.`usuario` (`Login` VARCHAR(25) NOT NULL , `Password` VARCHAR(50) NOT NULL , PRIMARY KEY (`Login`)) ENGINE = InnoDB;

CREATE TABLE `prueba1`.`tarea` (`ID` INT NOT NULL AUTO_INCREMENT , `Title` VARCHAR(50) NOT NULL , `Detail` VARCHAR(255) NOT NULL , `CreateDate` DATE NOT NULL , `LimitDate` DATE NULL , `Again` SMALLINT NULL , `Important` BOOLEAN NOT NULL , `User` VARCHAR(50) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = InnoDB;

ALTER TABLE `tarea` ADD FOREIGN KEY (`User`) REFERENCES `usuario`(`Login`) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE `prueba1`.`subtareas` (`ID` INT NOT NULL AUTO_INCREMENT , `Title` VARCHAR(50) NOT NULL , `Tarea` INT NOT NULL , PRIMARY KEY (`ID`)) ENGINE = InnoDB;

ALTER TABLE `subtareas` ADD FOREIGN KEY (`Tarea`) REFERENCES `tarea`(`ID`) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE `prueba1`.`lista` (`name` VARCHAR(50) NOT NULL , `tarea` INT NOT NULL , PRIMARY KEY (`name`)) ENGINE = InnoDB;

ALTER TABLE `lista` ADD FOREIGN KEY (`tarea`) REFERENCES `tarea`(`ID`) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE `prueba1`.`enemigo` (`user` VARCHAR(50) NOT NULL , `HP` INT NOT NULL , `type` BOOLEAN NOT NULL , `loot` INT NOT NULL , `damage` INT NOT NULL , PRIMARY KEY (`user`)) ENGINE = InnoDB;

ALTER TABLE `enemigo` ADD FOREIGN KEY (`user`) REFERENCES `usuario`(`Login`) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE `prueba1`.`personaje` (`user` VARCHAR(50) NOT NULL , `HP` INT NOT NULL , `action` INT NOT NULL , PRIMARY KEY (`user`)) ENGINE = InnoDB;

ALTER TABLE `personaje` ADD FOREIGN KEY (`user`) REFERENCES `usuario`(`Login`) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE `prueba1`.`objeto` (`name` VARCHAR(50) NOT NULL , `type` VARCHAR(8) NOT NULL , `ATK` INT NOT NULL , `DEF` INT NOT NULL , `HP` INT NOT NULL , `recoil` INT NOT NULL , `regeneration` INT NOT NULL , `unlocked` BOOLEAN NOT NULL , `equiped` BOOLEAN NOT NULL , `user` VARCHAR(50) NOT NULL , PRIMARY KEY (`name`)) ENGINE = InnoDB;

ALTER TABLE `objeto` ADD FOREIGN KEY (`user`) REFERENCES `usuario`(`Login`) ON DELETE CASCADE ON UPDATE CASCADE;
